# 3) file: `{root}/src/signal_sync/tools/stock_tool.py`

Place this file in your repo. It fetches and caches yfinance time series and returns a path to the cached JSON the agents can consume.

```python
# src/signal_sync/tools/stock_tool.py
import os
import json
from datetime import datetime
import yfinance as yf

ROOT = os.path.abspath(os.path.join(os.path.dirname(__file__), "..", "..", ".."))
CACHE_DIR = os.path.join(ROOT, "src", "signal_sync", "tools", "cache")
os.makedirs(CACHE_DIR, exist_ok=True)

def fetch_yfinance(ticker: str, period: str = "30d", interval: str = "1d") -> str:
    """
    Fetches historical data using yfinance, saves JSON cache and returns filepath.
    period examples: '7d','30d','60d', '1y'
    interval examples: '1d','1h'
    """
    ticker = ticker.upper()
    ts = yf.Ticker(ticker)
    # .history returns pandas DataFrame
    df = ts.history(period=period, interval=interval, actions=False)
    data = {
        "ticker": ticker,
        "period": period,
        "interval": interval,
        "fetched_at": datetime.utcnow().isoformat() + "Z",
        "data": df.reset_index().to_dict(orient="records")
    }
    fname = f"{ticker}_yfinance_{period}_{interval}_{datetime.utcnow().strftime('%Y%m%dT%H%M%SZ')}.json"
    path = os.path.join(CACHE_DIR, fname)
    with open(path, "w") as f:
        json.dump(data, f, default=str, indent=2)
    return path

# quick CLI test
if __name__ == "__main__":
    import sys
    ticker = sys.argv[1] if len(sys.argv) > 1 else "AAPL"
    print(fetch_yfinance(ticker, period="15d", interval="1d"))
```

**Notes**

* `yfinance` returns a pandas DataFrame; we convert to records for JSON.
* Agents should call `fetch_yfinance` and then analyze the cached JSON file path.

```python
import streamlit as st
import requests
import tempfile
import os
import json
from pathlib import Path
# We will call the local Python orchestrator directly; simpler: import the orchestrator module if same environment.
# If running separately, you can call an API endpoint. Here we assume same PYTHONPATH.

st.set_page_config(page_title="BGV Verification", layout="centered")
st.title("BGV — Background Verification")

company_name = st.text_input("Company name")
ticker = st.text_input("Ticker (exchange symbol)")
sector = st.text_input("Sector")
annual_pdf = st.file_uploader("Upload Annual Report PDF", type=["pdf"])

if st.button("Run BGV Crew"):
    if not company_name or not ticker or not annual_pdf:
        st.error("Please provide company name, ticker and annual report PDF.")
    else:
        # save uploaded file
        uploads_dir = Path("uploads")
        uploads_dir.mkdir(parents=True, exist_ok=True)
        pdf_path = uploads_dir / f"{ticker}_annual_report.pdf"
        with open(pdf_path, "wb") as f:
            f.write(annual_pdf.read())

        st.info("Triggering BGV orchestrator (this can take a minute)...")
        # If orchestrator is importable, call function directly:
        try:
            from src.signal_sync.bgv_orchestrator import run_bgv_pipeline
            out_path = run_bgv_pipeline(company_name.strip(), ticker.strip(), sector.strip(), str(pdf_path))
            st.success(f"BGV pipeline completed. Output: {out_path}")
            with open(out_path) as f:
                st.json(json.load(f))
        except Exception as e:
            st.error(f"Failed to run orchestrator locally: {e}")
            st.write("If you run the orchestrator as a separate service, POST the form data to its endpoint.")
```

# 5) Orchestrator + agent skeletons

Place under `{root}/src/signal_sync/`

`bgv_orchestrator.py` — coordinates agents and writes final `bgv_output.json`.

```python
# src/signal_sync/bgv_orchestrator.py
import os
import json
from datetime import datetime
from src.signal_sync.tools.stock_tool import fetch_yfinance
from src.signal_sync.agents.overview_agent import overview_agent_run
from src.signal_sync.agents.management_agent import management_agent_run
from src.signal_sync.agents.financial_agent import financial_agent_run
from src.signal_sync.agents.scam_agent import scam_agent_run

OUTPUT_DIR = os.path.join(os.path.dirname(__file__), "output_data")
os.makedirs(OUTPUT_DIR, exist_ok=True)
OUTPUT_FILE = os.path.join(OUTPUT_DIR, "bgv_output.json")

def run_bgv_pipeline(company_name: str, ticker: str, sector: str, annual_report_pdf_path: str) -> str:
    """
    Orchestrates the agents. Each agent returns a dict conforming to the JSON Schema fragments.
    """
    # 1. Overview agent (web search + LLM)
    overview = overview_agent_run(company_name, ticker, sector)

    # 2. Management research agent (people profiles, red flags)
    management = management_agent_run(company_name, ticker, sector)

    # 3. Financial irregularities agent: analyze annual report PDF + financial tables
    financial = financial_agent_run(annual_report_pdf_path, company_name, ticker)

    # 4. Fetch time-series data for the scam detection agent
    ts_path = fetch_yfinance(ticker, period="30d", interval="1d")

    # 5. Scam detection agent analyzes time-series
    scam = scam_agent_run(ts_path, company_name, ticker)

    # 6. Combine and score
    output = {
        "meta": {
            "generated_at": datetime.utcnow().isoformat() + "Z",
            "pipeline_version": "bgv_v0.1",
            "sources": overview.get("sources", []) + management.get("sources", [])
        },
        "company": overview.get("company", {}),
        "scores": {
            "trustworthiness_score": round((overview.get("trust_score", 60) + financial.get("fin_score", 60) + (100 - management.get("management_risk", 40))) / 3,2),
            "financial_integrity_score": financial.get("fin_score", 60),
            "management_risk_score": management.get("management_risk", 40),
            "market_manipulation_risk_score": scam.get("market_manipulation_risk", 10)
        },
        "findings": {
            "overview_findings": overview.get("findings", []),
            "management_findings": management.get("findings", []),
            "financial_irregularities": financial.get("findings", []),
            "scam_signals": scam.get("findings", [])
        },
        "evidence": {
            "documents": [
                {"type":"annual_report","location":annual_report_pdf_path,"summary":financial.get("report_summary","")}
            ],
            "time_series": [
                {"source":"yfinance","from":ts_path, "to":ts_path, "file":ts_path, "metrics": scam.get("metrics", {})}
            ],
            "people_profiles": management.get("people_profiles", [])
        },
        "final_verdict": "Moderate risk: follow-up recommended."
    }

    with open(OUTPUT_FILE, "w") as f:
        json.dump(output, f, indent=2, default=str)

    return OUTPUT_FILE
```

```python
# src/signal_sync/agents/scam_agent.py
import json
def scam_agent_run(ts_json_path: str, company_name: str, ticker: str) -> dict:
    """
    Load cached yfinance JSON and look for anomalies:
    - Large volume spikes (e.g > 3x median)
    - Price gaps
    - Volume unaccompanied by news (TODO: cross-check with SerperDev)
    """
    with open(ts_json_path) as f:
        data = json.load(f)

    records = data.get("data", [])
    volumes = [r.get("Volume") or r.get("volume") for r in records if r.get("Volume") or r.get("volume")]
    volumes = [v for v in volumes if isinstance(v, (int, float))]
    findings = []
    metrics = {}

    if volumes:
        import statistics
        med = statistics.median(volumes)
        max_v = max(volumes)
        if med > 0:
            ratio = max_v / med
            metrics["max_volume_spike"] = ratio
            if ratio > 3:
                findings.append(f"Detected volume spike: {ratio:.2f}x median volume")
    else:
        metrics["max_volume_spike"] = None

    return {"findings": findings, "market_manipulation_risk": 20 if findings else 5, "metrics": metrics}
```
